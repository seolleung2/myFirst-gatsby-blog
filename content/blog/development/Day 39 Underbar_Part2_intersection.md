  let arrNum = arrays.length

      transObj[lumpOfArrays[i]] = 1
    } else {
      transObj[lumpOfArrays[i]] = transObj[lumpOfArrays[i]] + 1
    }
  }
  let result = []

    if (transObj[key] === arrNum) {
      result.push(key)
    }
  }
  return result
}
